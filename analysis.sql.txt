-- Q1: Total Revenue
SELECT SUM(total_amount) AS total_revenue
FROM orders;

-- Q2: Orders per Customer
SELECT c.customer_name, COUNT(o.order_id) AS order_count
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name;

-- Q3: Top 3 Customers by Spending
SELECT c.customer_name, SUM(o.total_amount) AS spent
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name
ORDER BY spent DESC
LIMIT 3;

-- Q4: Most Sold Product
SELECT product_name, SUM(quantity) AS total_quantity
FROM order_items
GROUP BY product_name
ORDER BY total_quantity DESC
LIMIT 1;

-- Q5: Revenue by Category
SELECT category, SUM(quantity * price) AS revenue
FROM order_items
GROUP BY category
ORDER BY revenue DESC;

-- Q6: Highest Revenue City
SELECT c.city, SUM(o.total_amount) AS revenue
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.city
ORDER BY revenue DESC
LIMIT 1;

-- Q7: Repeat Customers
SELECT c.customer_name
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name
HAVING COUNT(o.order_id) > 1;

-- Q8: Customer Lifetime Value
SELECT c.customer_name, SUM(o.total_amount) AS clv
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name
ORDER BY clv DESC;

-- Q9: Monthly Revenue
SELECT DATE_FORMAT(order_date, '%M %Y') AS month_year,
       SUM(total_amount) AS revenue
FROM orders
GROUP BY month_year
ORDER BY MIN(order_date);

-- Q10: Running Total Revenue
SELECT order_date,
       SUM(total_amount) AS daily_revenue,
       SUM(SUM(total_amount)) OVER (ORDER BY order_date) AS running_total
FROM orders
GROUP BY order_date
ORDER BY order_date;

-- Q11: Rank Customers
SELECT c.customer_name,
       SUM(o.total_amount) AS total_spent,
       RANK() OVER (ORDER BY SUM(o.total_amount) DESC) AS ranking
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name;

-- Q12: Top 20% Customers
WITH customer_spending AS (
    SELECT c.customer_name,
           SUM(o.total_amount) AS total_spent,
           NTILE(5) OVER (ORDER BY SUM(o.total_amount) DESC) AS bucket
    FROM customers c
    JOIN orders o
    ON c.customer_id = o.customer_id
    GROUP BY c.customer_name
)
SELECT *
FROM customer_spending
WHERE bucket = 1;

-- Q13: Churn Customers
SELECT c.customer_name
FROM customers c
JOIN orders o
ON c.customer_id = o.customer_id
GROUP BY c.customer_name
HAVING MIN(o.order_date) = MAX(o.order_date);

-- Q14: Category Contribution %
SELECT category,
       SUM(quantity * price) AS revenue,
       ROUND((SUM(quantity * price) / 
       SUM(SUM(quantity * price)) OVER ()) * 100.0, 2) AS contribution_percentage
FROM order_items
GROUP BY category
ORDER BY revenue DESC;
